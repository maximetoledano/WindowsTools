
# Suppression des portion de texte entoures de balises conflictuelles
(.*)(\{.*\})(.*)=$1$3
(.*)(\(.*\))(.*)=$1$3

# Conversion explicite des chaines conflictuelles
(.*)(ASSVIE)(.*)=$1#ASSC-VIE#$3
(.*)(ASVIE)(.*)=$1#ASSC-VIE#$3

# Reconduction des nombres
(.*)\.([0-9]+)\.(.*)=$1#$2#$3
([0-9]+)\.(.*)=$1#$2
(.*)\.([0-9]+)=$1#$2
